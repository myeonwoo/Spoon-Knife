-- 선행 작업 (로컬 개발 머신 환경)
  npm, git

-- nuxt 프로젝트 생성 
  가이드: https://nuxtjs.org/docs/get-started/installation
  실행
    npm init nuxt-app <project-name>

      ? Continuous integration: None
      ? Version control system: Git

-- plugin 추가 v-calendar : https://vcalendar.io/installation.html
  실행
    cd <project-name>
    npm install v-calendar


  이미 작업된 파일 수정 적용됨 (git)
    - plugins/vue-calendar.js 추가
      import Vue from 'vue';
      import VCalendar from 'v-calendar';

      // Use v-calendar & v-date-picker components
      Vue.use(VCalendar, {
        componentPrefix: 'vc',
      });
    - nuxt.config.js 수정
      plugins: [
        {src:'~/plugins/vue-calendar.js', mode: 'client'}
      ],


-- Git 소스 적용
  git init .
  git remote add origin git@github.com:myeonwoo/mvc.git
  git fetch origin
  git branch -f master origin/master
  git reset .
  git checkout -f

-- 로컬 개발 서버 실행
  npm run dev
-- 개발 시작
